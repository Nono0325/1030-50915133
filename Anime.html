Anime.js 動畫展示

Anime.js 官網

這是一個炫麗的動畫效果展示，使用 Anime.js 創建 CSS 動畫效果。

範例程式碼


            <script src="https://cdnjs.cloudflare.com/ajax/libs/animejs/3.2.1/anime.min.js"></script>

            <button id="start-animation">開始動畫</button>
            <div id="boxes-container"></div>
            <script>
                document.getElementById('start-animation').onclick = function() {
                    var container = document.getElementById('boxes-container');
                    container.innerHTML = ''; // 清空容器
                    for (let i = 0; i < 5; i++) { // 创建5个方块
                        var box = document.createElement('div');
                        box.className = 'animated-box';
                        container.appendChild(box);

                        // 随机颜色
                        var randomColor = '#' + Math.floor(Math.random()*16777215).toString(16);
                        // 动画效果
                        anime({
                            targets: box,
                            translateX: 250,
                            rotate: '1turn',
                            scale: 1.5,
                            backgroundColor: randomColor, // 随机颜色变化
                            opacity: [0, 1], // 渐变效果
                            duration: 2000,
                            easing: 'easeInOutExpo'
                        });
                    }
                };
            </script>
