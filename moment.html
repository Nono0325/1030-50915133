Moment.js 日期展示

Moment.js 官網

使用 Moment.js 進行日期和時間處理。

範例程式碼


            <script src="https://cdn.jsdelivr.net/npm/moment@2.29.1/moment.min.js"></script>
            <script>
                // 當前日期和時間
                document.getElementById('moment-example').innerHTML = moment().format('LLLL');
                
                // 顯示其他日期格式
                const moreExamples = [
                    moment().format('YYYY-MM-DD'), // 簡單日期格式
                    moment().format('hh:mm A'), // 12小時制時間格式
                    moment().format('dddd'), // 星期幾
                    moment().format('MMM Do YYYY, h:mm:ss a'), // 自定義格式
                    moment('2025-01-01').diff(moment(), 'days'), // 日期差異
                    moment().add(1, 'months').format('YYYY-MM-DD'), // 一個月後的日期
                    moment().subtract(2, 'weeks').format('YYYY-MM-DD') // 兩週前的日期
                ];
                
                document.getElementById('more-examples').innerHTML = moreExamples.join('<br>');
            </script>
