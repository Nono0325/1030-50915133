Three.js 3D 圖形展示

Three.js 官網

使用 Three.js 創建多種 3D 圖形和動畫。

範例程式碼


const scene = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(75, window.innerWidth / 400, 0.1, 1000);
const renderer = new THREE.WebGLRenderer();
renderer.setSize(window.innerWidth, 400);
document.getElementById('three-container').appendChild(renderer.domElement);

// 創立方體
const cubeGeometry = new THREE.BoxGeometry();
const cubeMaterial = new THREE.MeshBasicMaterial({ color: 0x00ff00 });
const cube = new THREE.Mesh(cubeGeometry, cubeMaterial);
cube.position.x = -2; // 調整立方體位置
scene.add(cube);

// 創造球體
const sphereGeometry = new THREE.SphereGeometry(1, 32, 32);
const sphereMaterial = new THREE.MeshBasicMaterial({ color: 0xff0000 });
const sphere = new THREE.Mesh(sphereGeometry, sphereMaterial);
sphere.position.x = 0; // 調整球體位置
scene.add(sphere);

// 創造圓柱體
const cylinderGeometry = new THREE.CylinderGeometry(0.5, 0.5, 2, 32);
const cylinderMaterial = new THREE.MeshBasicMaterial({ color: 0x0000ff });
const cylinder = new THREE.Mesh(cylinderGeometry, cylinderMaterial);
cylinder.position.x = 2; // 調整圓柱體位置
scene.add(cylinder);

// 設定相機位置
camera.position.z = 5;

// 動畫函數
function animate() {
    requestAnimationFrame(animate);
    cube.rotation.x += 0.01; // 立方體旋轉
    cube.rotation.y += 0.01; // 立方體旋轉

    sphere.rotation.x += 0.02; // 球體旋轉
    sphere.rotation.y += 0.02; // 球體旋轉

    cylinder.rotation.x += 0.03; // 圓柱體旋轉
    cylinder.rotation.y += 0.03; // 圓柱體旋轉

    renderer.render(scene, camera); // 渲染場景
}
animate();
